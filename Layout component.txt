üì± Responsive Grid Component

```blade
{{-- resources/views/components/layouts/responsive-grid.blade.php --}}
@props([
    'columns' => [
        'default' => 1,
        'sm' => 2, 
        'md' => 3,
        'lg' => 4,
        'xl' => 5
    ],
    'gap' => '4'
])

@php
    $gapClasses = [
        '2' => 'gap-2',
        '4' => 'gap-4',
        '6' => 'gap-6',
        '8' => 'gap-8'
    ];
    
    $columnClasses = "grid grid-cols-{$columns['default']} {$gapClasses[$gap]}";
    
    foreach($columns as $breakpoint => $cols) {
        if($breakpoint !== 'default') {
            $columnClasses .= " {$breakpoint}:grid-cols-{$cols}";
        }
    }
@endphp

<div {{ $attributes->merge(['class' => $columnClasses]) }}>
    {{ $slot }}
</div>
```
